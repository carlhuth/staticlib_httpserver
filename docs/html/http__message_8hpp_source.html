844  template <typename DictionaryType>
845  inline static void change_value(DictionaryType& dict, const std::string& key,
846  const std::string& value) {
847  // retrieve all current values for key
848  std::pair<typename DictionaryType::iterator, typename DictionaryType::iterator>
849  result_pair = dict.equal_range(key);
850  if (result_pair.first == dict.end()) {
851  // no values exist -> add a new key
852  dict.insert(std::make_pair(key, value));
853  } else {
854  // set the first value found for the key to the new one
855  result_pair.first->second = value;
856  // remove any remaining values
857  typename DictionaryType::iterator i;
858  ++(result_pair.first);
859  while (result_pair.first != result_pair.second) {
860  i = result_pair.first;
861  ++(result_pair.first);
862  dict.erase(i);
863  }
864  }
865  }
866 
873  template <typename DictionaryType>
874  inline static void delete_value(DictionaryType& dict, const std::string& key) {
875  std::pair<typename DictionaryType::iterator, typename DictionaryType::iterator>
876  result_pair = dict.equal_range(key);
877  if (result_pair.first != dict.end()) {
878  dict.erase(result_pair.first, result_pair.second);
879  }
880  }
