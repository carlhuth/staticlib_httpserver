795 inline bool parser::is_special(int c) {
796  switch (c) {
797  case '(': case ')': case '<': case '>': case '@':
798  case ',': case ';': case ':': case '\\': case '"':
799  case '/': case '[': case ']': case '?': case '=':
800  case '{': case '}': case ' ': case '\t':
801  return true;
802  default:
803  return false;
804  }
805 }

815 inline bool parser::is_cookie_attribute(const std::string& name, bool set_cookie_header) {
816  return (name.empty() || name[0] == '$' || (set_cookie_header &&
817  (
818  // This is needed because of a very lenient determination in parse_cookie_header() of what
819  // qualifies as a cookie-pair in a Set-Cookie header.
820  // According to RFC 6265, everything after the first semicolon is a cookie attribute, but RFC 2109,
821  // which is obsolete, allowed multiple comma separated cookies.
822  // parse_cookie_header() is very conservatively assuming that any <name>=<value> pair in a
823  // Set-Cookie header is a cookie-pair unless <name> is a known cookie attribute.
824  pion::algorithm::iequals(name, "Comment")
825  || pion::algorithm::iequals(name, "Domain")
826  || pion::algorithm::iequals(name, "Max-Age")
827  || pion::algorithm::iequals(name, "Path")
828  || pion::algorithm::iequals(name, "Secure")
829  || pion::algorithm::iequals(name, "Version")
830  || pion::algorithm::iequals(name, "Expires")
831  || pion::algorithm::iequals(name, "HttpOnly")
832  )
833  ));
834 }
