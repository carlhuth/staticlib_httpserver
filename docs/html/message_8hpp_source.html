849  template <typename DictionaryType>
850  inline static void change_value(DictionaryType& dict, const std::string& key,
851  const std::string& value) {
852  // retrieve all current values for key
853  std::pair<typename DictionaryType::iterator, typename DictionaryType::iterator>
854  result_pair = dict.equal_range(key);
855  if (result_pair.first == dict.end()) {
856  // no values exist -> add a new key
857  dict.insert(std::make_pair(key, value));
858  } else {
859  // set the first value found for the key to the new one
860  result_pair.first->second = value;
861  // remove any remaining values
862  typename DictionaryType::iterator i;
863  ++(result_pair.first);
864  while (result_pair.first != result_pair.second) {
865  i = result_pair.first;
866  ++(result_pair.first);
867  dict.erase(i);
868  }
869  }
870  }
871 
878  template <typename DictionaryType>
879  inline static void delete_value(DictionaryType& dict, const std::string& key) {
880  std::pair<typename DictionaryType::iterator, typename DictionaryType::iterator>
881  result_pair = dict.equal_range(key);
882  if (result_pair.first != dict.end()) {
883  dict.erase(result_pair.first, result_pair.second);
884  }
885  }
