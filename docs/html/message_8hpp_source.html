746  template <typename DictionaryType>
747  inline static void change_value(DictionaryType& dict, const std::string& key,
748  const std::string& value) {
749  // retrieve all current values for key
750  std::pair<typename DictionaryType::iterator, typename DictionaryType::iterator>
751  result_pair = dict.equal_range(key);
752  if (result_pair.first == dict.end()) {
753  // no values exist -> add a new key
754  dict.insert(std::make_pair(key, value));
755  } else {
756  // set the first value found for the key to the new one
757  result_pair.first->second = value;
758  // remove any remaining values
759  typename DictionaryType::iterator i;
760  ++(result_pair.first);
761  while (result_pair.first != result_pair.second) {
762  i = result_pair.first;
763  ++(result_pair.first);
764  dict.erase(i);
765  }
766  }
767  }
768 
775  template <typename DictionaryType>
776  inline static void delete_value(DictionaryType& dict, const std::string& key) {
777  std::pair<typename DictionaryType::iterator, typename DictionaryType::iterator>
778  result_pair = dict.equal_range(key);
779  if (result_pair.first != dict.end()) {
780  dict.erase(result_pair.first, result_pair.second);
781  }
782  }
