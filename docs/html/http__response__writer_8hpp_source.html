415  void send_more_data(const bool send_final_chunk, SendHandler send_handler) {
416  // make sure that we did not lose the TCP connection
417  if (m_tcp_conn->is_open()) {
418  // make sure that the content-length is up-to-date
419  flush_content_stream();
420  // prepare the write buffers to be sent

422  prepare_write_buffers(write_buffers, send_final_chunk);
423  // send data in the write buffers
